Proteção Cross-Tenant — TenantModel + Testes Automatizados
app/Models/Concerns · tests/Feature/Security
Laravel · PHPUnit
Por que isso é crítico: Em um Shared Database, um único Review::all() sem filtro expõe dados de todos os tenants. A solução é uma classe base que injeta o tenant_id automaticamente em toda query, tornando o vazamento impossível por design — não por disciplina do desenvolvedor. Os testes de cross-tenant garantem que o isolamento funciona mesmo após refatorações futuras.
app/Models/TenantModel.php PHP 
<?php

namespace App\Models;

use Illuminate\Database\Eloquent\Model;
use App\Models\Concerns\HasTenant;
use App\Models\Scopes\TenantScope;

/**
 * Classe base para todos os modelos que pertencem a um tenant.
 * Herdar desta classe é OBRIGATÓRIO para qualquer model
 * que tenha coluna tenant_id.
 *
 * Garante:
 *  1. TenantScope aplicado em TODAS as queries (SELECT, UPDATE, DELETE)
 *  2. tenant_id injetado automaticamente em CREATE
 *  3. Impossibilidade de salvar/atualizar registro sem tenant_id válido
 */
abstract class TenantModel extends Model
{
    use HasTenant;

    /**
     * Boot: registra o TenantScope globalmente.
     * Roda uma vez por request, não por query.
     */
    protected static function boot(): void
    {
        parent::boot();

        // Aplica filtro de tenant em todas as queries SELECT/UPDATE/DELETE
        static::addGlobalScope(new TenantScope);

        // Injeta tenant_id automaticamente ao criar qualquer registro
        static::creating(function (Model $model) {
            if (empty($model->tenant_id)) {
                $tenantId = app('current.tenant.id');

                if (! $tenantId) {
                    throw new \RuntimeException(
                        'Tentativa de criar registro sem tenant_id resolvido. ' .
                        'Verifique se o TenantMiddleware está na rota.'
                    );
                }

                $model->tenant_id = $tenantId;
            }
        });
    }

    /**
     * Desabilita o TenantScope para queries administrativas.
     * USO RESTRITO: apenas em Jobs de relatório e painel Super Admin.
     * NUNCA use em rotas de lojista.
     */
    public static function withoutTenantScope(): Builder
    {
        return static::withoutGlobalScope(TenantScope::class);
    }
}
app/Models/Concerns/HasTenant.php PHP 
<?php

namespace App\Models\Concerns;

use App\Models\Tenant;

/**
 * Trait reutilizável para qualquer model que precise
 * do relacionamento com Tenant sem herdar TenantModel.
 * Ex: User tem tenant_id mas também é usado pelo Super Admin.
 */
trait HasTenant
{
    /**
     * Relacionamento: cada model pertence a um Tenant.
     */
    public function tenant(): BelongsTo
    {
        return $this->belongsTo(Tenant::class, 'tenant_id');
    }

    /**
     * Scope para forçar um tenant específico (uso em testes).
     * Ex: Review::forTenant($uuid)->get()
     */
    public function scopeForTenant(Builder $query, string $tenantId): Builder
    {
        return $query->where('tenant_id', $tenantId);
    }

    /**
     * Verifica se o modelo pertence ao tenant atual.
     * Use em Policies: $this->belongsToCurrentTenant()
     */
    public function belongsToCurrentTenant(): bool
    {
        return $this->tenant_id === app('current.tenant.id');
    }
}
app/Models/Scopes/TenantScope.php PHP 
<?php

namespace App\Models\Scopes;

use Illuminate\Database\Eloquent\{Builder, Model, Scope};

class TenantScope implements Scope
{
    /**
     * Injeta automaticamente WHERE tenant_id = :current
     * em toda query que usa qualquer TenantModel.
     *
     * Laravel aplica isso em: SELECT, UPDATE, DELETE.
     * Não se aplica em INSERT (tratado no boot do TenantModel).
     */
    public function apply(Builder $builder, Model $model): void
    {
        $tenantId = app('current.tenant.id');

        // Se não houver tenant resolvido (ex: Jobs do Super Admin),
        // não aplica o filtro. O Job deve chamar withoutTenantScope()
        // explicitamente para deixar a intenção clara no código.
        if ($tenantId) {
            $builder->where(
                $model->getTable() . '.tenant_id',
                $tenantId
            );
        }
    }
}

/**

class CrossTenantAccessTest extends TestCase

    use RefreshDatabase;

    private Tenant $tenantA;
    private Tenant $tenantB;
    private User   $ownerA;

    protected function setUp(): void
    {
        parent::setUp();

        // Cria dois tenants independentes
        $this->tenantA = Tenant::factory()->create();
        $this->tenantB = Tenant::factory()->create();

        // Cria avaliações para cada tenant
        Review::factory()->count(5)->create(['tenant_id' => $this->tenantA->id]);
        Review::factory()->count(3)->create(['tenant_id' => $this->tenantB->id]);

        // Usuário (lojista) do Tenant A
        $this->ownerA = User::factory()->create([
            'tenant_id' => $this->tenantA->id,
            'role'      => 'owner',
        ]);
    }

    /** @test */
    public function lojista_nao_ve_avaliacoes_de_outro_tenant(): void
    {
        // Simula lojista A autenticado
        $this->actingAs($this->ownerA);
        app()->instance('current.tenant.id', $this->tenantA->id);

        // Tenta buscar TODAS as reviews sem filtro explícito
        $reviews = Review::all();

        // Deve retornar APENAS as 5 reviews do tenant A
        $this->assertCount(5, $reviews);
        $reviews->each(fn($r) => $this->assertEquals($this->tenantA->id, $r->tenant_id));
    }

    /** @test */
    public function api_rejeita_review_com_slug_de_outro_tenant(): void
    {
        // Tenta POST com slug do tenant B enquanto autenticado como A
        $response = $this
            ->actingAs($this->ownerA, 'sanctum')
            ->postJson('/api/review', [
                'slug'   => $this->tenantB->slug, // slug de OUTRO tenant
                'rating' => 5,
            ]);

        // API deve rejeitar — tenant não pertence ao usuário autenticado
        $response->assertStatus(404); // 404 e não 403 — não revelamos que existe
    }

        // Tenta acessar diretamente a review do tenant B pela URL
        $response = $this->getJson("/painel/avaliacoes/{$reviewB->id}");

app/Http/Middleware/ResolveTenant.php PHP 
<?php

namespace App\Http\Middleware;

use Closure;
use App\Models\Tenant;
use Illuminate\Http\Request;

/**
 * Resolve o tenant atual e registra no container.
 *
 * Para rotas de API pública (/api/review):
 *   Resolve pelo ?slug= ou pelo {slug} na URL.
 *
 * Para rotas do painel do lojista (/painel/*):
 *   Resolve pelo tenant_id do usuário autenticado.
 *
 * Para rotas do Super Admin (/admin/*):
 *   Não resolve — super admin opera sem filtro de tenant.
 */
class ResolveTenant
{
    public function handle(Request $request, Closure $next, string $source = 'auth')
    {
        $tenantId = match ($source) {
            // Painel do lojista: usa o tenant do usuário logado
            'auth' => auth()->user()?->tenant_id,

            // API pública: resolve pelo slug na URL
            'slug' => Tenant::where('slug', $request->route('slug') ?? $request->input('slug'))
                          ->where('active', true)
                          ->value('id'),

            default => null,
        };

        if (! $tenantId) {
            return response()->json(['error' => 'not_found'], 404);
        }

        // Registra no container — acessível em qualquer lugar via app('current.tenant.id')
        app()->instance('current.tenant.id', $tenantId);

        return $next($request);
    }
}

<?php

namespace App\Services\Notification;

use App\Services\Notification\Contracts\WhatsAppDriver;
use Illuminate\Support\Facades\{Http, Log};

/**
 * Driver Evolution API — para MVP e até ~200 tenants.
 * Vantagem: barato, sem aprovação Meta.
 * Risco: unofficial API, risco de banimento em alto volume.
 * Migrar para MetaCloudDriver ao passar de 200 tenants ativos.
 */
class EvolutionDriver implements WhatsAppDriver
{
    public function __construct(
        private readonly string $baseUrl,
        private readonly string $apiKey,
        private readonly string $instance,
    ) {}

    public function send(string $to, string $message): bool
    {
        try {
            $response = Http::withHeaders([
                'apikey'       => $this->apiKey,
                'Content-Type' => 'application/json',
            ])
            ->timeout(8)
            ->post("{$this->baseUrl}/message/sendText/{$this->instance}", [
                'number'  => preg_replace('/\D/', '', $to), // só dígitos
                'options' => ['delay' => 0],
                'textMessage' => ['text' => $message],
            ]);

            return $response->successful();

        } catch (\Throwable $e) {
            Log::error('[EvolutionDriver] Falha ao enviar', [
                'to'    => substr($to, 0, 6) . '****', // não loga número completo
                'error' => $e->getMessage(),
            ]);
            return false;
        }
    }

    public function healthCheck(): bool
    {
        try {
            return Http::withHeaders(['apikey' => $this->apiKey])
                ->timeout(5)
                ->get("{$this->baseUrl}/instance/connectionState/{$this->instance}")
                ->json('state') === 'open';
        } catch (\Throwable) {
            return false;
        }
    }

    public function driverName(): string { return 'evolution'; }
}

<?php

namespace App\Services\Notification;

use App\Services\Notification\Contracts\WhatsAppDriver;
use Illuminate\Support\Facades\{Http, Log};

/**
 * Driver Meta Cloud API (oficial).
 * Ativar quando: Evolution estiver causando banimentos
 * ou ao passar de 200 tenants ativos.
 *
 * Para ativar: config/services.php → whatsapp.driver = 'meta'
 * O resto do sistema não muda nada.
 */
class MetaCloudDriver implements WhatsAppDriver
{
    private const API_URL = 'https://graph.facebook.com/v19.0';

    public function __construct(
        private readonly string $phoneNumberId,
        private readonly string $accessToken,
    ) {}

    public function send(string $to, string $message): bool
    {
        try {
            $response = Http::withToken($this->accessToken)
                ->timeout(10)
                ->post(self::API_URL . "/{$this->phoneNumberId}/messages", [
                    'messaging_product' => 'whatsapp',
                    'to'                => preg_replace('/\D/', '', $to),
                    'type'              => 'text',
                    'text'              => ['body' => $message],
                ]);

            return $response->successful();

        } catch (\Throwable $e) {
            Log::error('[MetaCloudDriver] Falha', ['error' => $e->getMessage()]);
            return false;
        }
    }

    public function healthCheck(): bool
    {
        return Http::withToken($this->accessToken)
            ->get(self::API_URL . "/{$this->phoneNumberId}")
            ->successful();
    }

    public function driverName(): string { return 'meta_cloud'; }
}
app/Services/NotificationService.php PHP · Fallback Triplo 
<?php

namespace App\Services;

use App\Models\{Review, NotificationLog, Tenant};
use App\Services\Notification\Contracts\WhatsAppDriver;
use App\Services\Notification\LineDriver;
use Illuminate\Support\Facades\Mail;

class NotificationService
{
    public function __construct(
        private readonly WhatsAppDriver $whatsapp,
        private readonly LineDriver     $line,
    ) {}

    /**
     * Notifica lojista sobre avaliação negativa.
     *
     * Ordem de tentativa:
     *  1. Canal preferido do tenant (whatsapp | line)
     *  2. Se falhar 3x → fallback e-mail
     *  3. Log de todas as tentativas em notifications_log
     */
    public function notifyLowRating(Review $review): void
    {
        $tenant  = $review->tenant;
        $message = $this->buildMessage($review, $tenant);
        $sent    = false;
        $attempt = 0;

        while (! $sent && $attempt < 3) {
            $attempt++;

            $sent = match ($tenant->notify_channel) {
                'whatsapp' => $this->whatsapp->send($tenant->notify_target, $message),
                'line'     => $this->line->send($tenant->notify_target, $message),
                default    => false,
            };

            $this->logAttempt($review, $tenant->notify_channel, $sent, $attempt);

            if (! $sent) sleep(2); // backoff simples entre tentativas
        }

        // Fallback: após 3 falhas no canal principal, envia e-mail
        if (! $sent) {
            $this->fallbackEmail($review, $tenant);
        }
    }

    private function buildMessage(Review $review, Tenant $tenant): string
    {
        $stars    = str_repeat('⭐', $review->rating);
        $category = $review->issue_category ?? 'Não informado';
        $link     = route('painel.reviews.show', $review->id);

        return $tenant->locale === 'ja'
            ? "【CP Review】低評価が届きました\n評価: {$stars}\nカテゴリ: {$category}\n詳細: {$link}"
            : "⚠️ Nova avaliação negativa!\nNota: {$stars}\nProblema: {$category}\nVer detalhes: {$link}";
    }

    private function logAttempt(Review $review, string $channel, bool $success, int $attempt): void
    {
        NotificationLog::withoutTenantScope()->create([
            'review_id' => $review->id,
            'tenant_id' => $review->tenant_id,
            'channel'   => $channel,
            'status'    => $success ? 'sent' : ($attempt >= 3 ? 'failed' : 'retry'),
            'attempts'  => $attempt,
            'sent_at'   => $success ? now() : null,
        ]);
    }
}
app/Jobs/NotifyLowRatingJob.php PHP · Async Queue 
<?php

namespace App\Jobs;

use App\Models\Review;
use App\Services\NotificationService;
use Illuminate\Bus\Queueable;
use Illuminate\Contracts\Queue\ShouldQueue;
use Illuminate\Queue\{InteractsWithQueue, SerializesModels};
use Illuminate\Queue\Middleware\RateLimited;

class NotifyLowRatingJob implements ShouldQueue
{
    use Queueable, InteractsWithQueue, SerializesModels;

    /**
     * Tentativas máximas antes de descartar o job.
     * Com backoff exponencial: 1s, 4s, 16s entre tentativas.
     */
    public int $tries = 3;
    public int $backoff = 60; // segundos entre tentativas do worker

    public function __construct(private readonly Review $review) {}

    public function handle(NotificationService $service): void
    {
        // Verifica se o tenant ainda está ativo antes de notificar
        if (! $this->review->tenant?->active) {
            return; // descarta silenciosamente
        }

        $service->notifyLowRating($this->review);
    }

    // Rate limit: máx 30 notificações por minuto por tenant
    public function middleware(): array
    {
        return [new RateLimited('notify:tenant:' . $this->review->tenant_id)];
    }
}

Pré-Agregação de BI — daily_metrics_summary + Heatmap
database/migrations · app/Console/Commands · app/Services/BiService
MySQL · Laravel Scheduler
O problema real: Com 500 tenants × 100 avaliações/dia = 50.000 registros/dia. Em 1 ano: 18 milhões de registros. Uma query de heatmap rodando GROUP BY created_at sobre essa tabela mata o banco em 2 segundos. A solução é um job que roda toda madrugada, consolida os dados do dia anterior numa tabela de métricas, e o dashboard lê dessa tabela pequena — não da tabela de avaliações.

    {

            // Chave composta: tenant + data + período
            $table->uuid('tenant_id');
            $table->date('metric_date');
            $table->enum('period', ['lunch', 'dinner', 'other', 'all']);

            // Contagens por faixa de nota
            $table->unsignedSmallInteger('total_reviews')->default(0);
            $table->unsignedSmallInteger('positive_count')->default(0); // 4-5
            $table->unsignedSmallInteger('negative_count')->default(0); // 1-3
            $table->unsignedSmallInteger('rating_1')->default(0);
            $table->unsignedSmallInteger('rating_2')->default(0);
            $table->unsignedSmallInteger('rating_3')->default(0);
            $table->unsignedSmallInteger('rating_4')->default(0);
            $table->unsignedSmallInteger('rating_5')->default(0);

            $table->decimal('avg_rating', 3, 2)->default(0);

            $table->unsignedSmallInteger('first_visit_count')->default(0);

app/Console/Commands/AggregateDailyMetrics.php PHP · Console Command 
<?php

namespace App\Console\Commands;

use Illuminate\Console\Command;
use App\Models\{Tenant, Review, DailyMetricsSummary};
use Illuminate\Support\Carbon;

class AggregateDailyMetrics extends Command
{
    protected $signature   = 'metrics:aggregate {--date= : Data no formato Y-m-d (padrão: ontem)}';
    protected $description = 'Agrega métricas diárias de avaliações na tabela daily_metrics_summary';

    public function handle(): int
    {
        $date = $this->option('date')
            ? Carbon::parse($this->option('date'))
            : Carbon::yesterday();

        $this->info("Agregando métricas para {$date->toDateString()}...");

        // Processa todos os tenants ativos
        Tenant::where('active', true)
            ->chunkById(50, function ($tenants) use ($date) {
                foreach ($tenants as $tenant) {
                    $this->aggregateTenant($tenant, $date);
                }
            });

        $this->info('Agregação concluída!');
        return Command::SUCCESS;
    }

    private function aggregateTenant(Tenant $tenant, Carbon $date): void
    {
        $periods = ['lunch', 'dinner', 'other', 'all'];

        foreach ($periods as $period) {
            $query = Review::withoutTenantScope()
                ->where('tenant_id', $tenant->id)
                ->whereDate('created_at', $date);

            if ($period !== 'all') {
                $query->where('visit_period', $period);
            }

            $reviews = $query->get([
                'rating', 'is_first_visit', 'issue_category'
            ]);

            if ($reviews->isEmpty()) continue;

            // Calcula top issues do dia
            $topIssues = $reviews
                ->whereNotNull('issue_category')
                ->groupBy('issue_category')
                ->map(fn($g) => ['category' => $g->first()->issue_category, 'count' => $g->count()])
                ->sortByDesc('count')
                ->take(3)
                ->values();

            // Upsert: recalcula se o job rodar duas vezes no mesmo dia
            DailyMetricsSummary::upsert([
                'tenant_id'        => $tenant->id,
                'metric_date'      => $date->toDateString(),
                'period'           => $period,
                'total_reviews'    => $reviews->count(),
                'positive_count'   => $reviews->where('rating', '>=', 4)->count(),
                'negative_count'   => $reviews->where('rating', '<=', 3)->count(),
                'rating_1'         => $reviews->where('rating', 1)->count(),
                'rating_2'         => $reviews->where('rating', 2)->count(),
                'rating_3'         => $reviews->where('rating', 3)->count(),
                'rating_4'         => $reviews->where('rating', 4)->count(),
                'rating_5'         => $reviews->where('rating', 5)->count(),
                'avg_rating'       => round($reviews->avg('rating'), 2),
                'first_visit_count'=> $reviews->where('is_first_visit', true)->count(),
                'returning_count'  => $reviews->where('is_first_visit', false)->count(),
                'top_issues'       => $topIssues->toJson(),
                'aggregated_at'    => now(),
            ], ['tenant_id', 'metric_date', 'period']);
        }
    }
}

<?php

namespace App\Services;

use App\Models\DailyMetricsSummary;
use Illuminate\Support\Collection;

class BiService
{
    /**
     * Retorna dados para o Heatmap (dias × períodos).
     *
     * Query simples e rápida: lê da tabela pré-agregada.
     * Nenhuma agregação em runtime. <5ms mesmo com 500 tenants.
     *
     * Retorna array compatível com qualquer lib de heatmap (JS/Chart.js):
     * [
     *   ['date' => '2026-04-01', 'period' => 'lunch', 'avg' => 4.2, 'total' => 12],
     *   ...
     * ]
     */
    public function heatmapData(string $tenantId, int $days = 30): Collection
    {
        return DailyMetricsSummary::where('tenant_id', $tenantId)
            ->where('metric_date', '>=', now()->subDays($days))
            ->where('period', '!=', 'all') // heatmap usa períodos separados
            ->orderBy('metric_date')
            ->get(['metric_date', 'period', 'avg_rating', 'negative_count', 'total_reviews'])
            ->map(fn($m) => [
                'date'     => $m->metric_date,
                'period'   => $m->period,
                'avg'      => (float) $m->avg_rating,
                'total'    => $m->total_reviews,
                'negative' => $m->negative_count,
                // intensidade de 0-1 para o heatmap (0 = verde, 1 = vermelho)
                'intensity' => $m->total_reviews > 0
                    ? round($m->negative_count / $m->total_reviews, 2)
                    : 0,
            ]);
    }

    /**

     * Lê da tabela pré-agregada com period='all'.

    public function monthlySummary(string $tenantId): array
    {
        $rows = DailyMetricsSummary::where('tenant_id', $tenantId)
            ->where('period', 'all')
            ->where('metric_date', '>=', now()->startOfMonth())
            ->get();

        return [
            'total'     => $rows->sum('total_reviews'),
            'positive'  => $rows->sum('positive_count'),
            'negative'  => $rows->sum('negative_count'),
            'avg_rating' => $rows->avg('avg_rating'),
            'new_clients'=> $rows->sum('first_visit_count'),
        ];
    }

public/pwa/offline-queue.js JS · IndexedDB Queue 
/**
 * OfflineQueue — fila persistente de uploads em IndexedDB.
 *
 * Por que IndexedDB e não localStorage?
 *  - localStorage é síncrono e limitado a 5MB
 *  - IndexedDB é assíncrono e suporta blobs (arquivos binários)
 *  - Fotos podem ter até 5MB — localStorage quebraria
 */

const OfflineQueue = {
  DB_NAME: 'cp_review_queue',
  DB_VERSION: 1,
  STORE: 'pending_uploads',
  _db: null,

  /** Abre (ou cria) o banco IndexedDB */
  async open() {
    if (this._db) return this._db;

    return new Promise((resolve, reject) => {
      const req = indexedDB.open(this.DB_NAME, this.DB_VERSION);

      req.onupgradeneeded = (e) => {
        const db = e.target.result;
        if (!db.objectStoreNames.contains(this.STORE)) {
          const store = db.createObjectStore(this.STORE, { keyPath: 'id', autoIncrement: true });
          store.createIndex('status', 'status', { unique: false });
        }
      };

      req.onsuccess  = (e) => { this._db = e.target.result; resolve(this._db); };
      req.onerror    = () => reject(req.error);
    });
  },

  /** Adiciona item à fila. Retorna o ID gerado. */
  async enqueue({ reviewToken, slug, photoBlob, photoName }) {
    const db = await this.open();
    return new Promise((resolve, reject) => {
      const tx  = db.transaction(this.STORE, 'readwrite');
      const req = tx.objectStore(this.STORE).add({
        reviewToken, slug, photoBlob, photoName,
        status:    'pending',  // pending | uploading | done | failed
        attempts:  0,
        createdAt: new Date().toISOString(),
      });
      req.onsuccess = () => resolve(req.result);
      req.onerror   = () => reject(req.error);
    });
  },

  /** Atualiza o status de um item na fila */
  async updateStatus(id, status, extras = {}) {
    const db = await this.open();
    return new Promise((resolve, reject) => {
      const tx    = db.transaction(this.STORE, 'readwrite');
      const store = tx.objectStore(this.STORE);
      const getReq = store.get(id);
      getReq.onsuccess = () => {
        const item = { ...getReq.result, status, ...extras };
        const putReq = store.put(item);
        putReq.onsuccess = () => resolve(item);
        putReq.onerror   = () => reject(putReq.error);
      };
    });
  },

  /** Busca todos os itens pendentes (para retry automático) */
  async getPending() {
    const db = await this.open();
    return new Promise((resolve) => {
      const tx    = db.transaction(this.STORE, 'readonly');
      const index = tx.objectStore(this.STORE).index('status');
      const req   = index.getAll(IDBKeyRange.only('pending'));
      req.onsuccess = () => resolve(req.result);
    });
  },
};
public/pwa/photo-upload.js JS · Upload com Queue 
/**
 * PhotoUploader — gerencia o upload de fotos com:
 *  - Compressão client-side (reduz 5MB para ~500KB antes de enviar)
 *  - Persistência na fila IndexedDB
 *  - Retry automático e manual
 *  - Callbacks de progresso para atualizar a UI do chat
 */

const PhotoUploader = {

  /** Comprime imagem no cliente antes de enviar.
   *  Mantém qualidade visual mas reduz tamanho ~80%. */
  async compress(file, maxWidth = 1200, quality = 0.8) {
    return new Promise((resolve) => {
      const img = new Image();
      img.onload = () => {
        const scale  = Math.min(1, maxWidth / img.width);
        const canvas = Object.assign(document.createElement('canvas'), {
          width:  img.width  * scale,
          height: img.height * scale,
        });
        canvas.getContext('2d').drawImage(img, 0, 0, canvas.width, canvas.height);
        canvas.toBlob(resolve, 'image/jpeg', quality);
      };
      img.src = URL.createObjectURL(file);
    });
  },

  /** Tenta enviar a foto. Persiste na fila se offline.
   *  onProgress(status) é chamado para atualizar a UI. */
  async upload({ file, reviewToken, slug, onProgress }) {
    // 1. Comprimir antes de qualquer coisa
    const compressed = await this.compress(file);
    const queueId    = await OfflineQueue.enqueue({
      reviewToken, slug,
      photoBlob: compressed,
      photoName: file.name,
    });

    onProgress?.({ status: 'uploading', queueId });

    // 2. Se offline, não tenta — espera o Service Worker sincronizar
    if (!navigator.onLine) {
      await OfflineQueue.updateStatus(queueId, 'pending');
      onProgress?.({ status: 'offline', queueId });
      return { success: false, offline: true, queueId };
    }

    // 3. Tenta enviar com timeout de 30s (3G lento)
    try {
      const fd = new FormData();
      fd.append('photo', compressed, `photo_${reviewToken}.jpg`);
      fd.append('review_token', reviewToken);
      fd.append('slug', slug);

      const ctrl = new AbortController();
      setTimeout(() => ctrl.abort(), 30000); // 30s timeout

      const res = await fetch('/api/media/upload', {
        method: 'POST', body: fd, signal: ctrl.signal
      });

      if (res.ok) {
        await OfflineQueue.updateStatus(queueId, 'done');
        onProgress?.({ status: 'done', queueId });
        return { success: true, queueId };
      }

      throw new Error(`HTTP ${res.status}`);

    } catch (err) {
      await OfflineQueue.updateStatus(queueId, 'failed', { error: err.message });
      onProgress?.({ status: 'failed', queueId });
      return { success: false, offline: false, queueId };
    }
  },
};

/**
 * Widget de foto com retry UI — integra no fluxo conversacional.
 * Mostra status visual: comprimindo → enviando → ✅ enviado / ⚠️ falhou
 * Botão de retry aparece apenas se falhar. Sem poluir o fluxo feliz.
 */

async function addPhotoWidget(reviewToken, slug) {
  const widgetHtml = `
    <div class="photo-widget" id="photo-widget">
      <label class="photo-label">
        <span>📷</span>
        <span>Adicionar foto</span>
        <span class="photo-opt">opcional</span>
        <input type="file" accept="image/*" id="photo-input" style="display:none">
      </label>
      <div class="photo-status" id="photo-status" style="display:none"></div>
    </div>
  `;

  await addWidget(widgetHtml, 200);

  document.getElementById('photo-input')?.addEventListener('change', async (e) => {
    const file      = e.target.files[0];
    if (!file) return;
    if (file.size > 10 * 1024 * 1024) { // 10MB hard limit
      setPhotoStatus('error', 'Foto muito grande. Máximo: 10MB');
      return;
    }

    setPhotoStatus('uploading', 'Comprimindo e enviando...');

    const result = await PhotoUploader.upload({
      file, reviewToken, slug,
      onProgress({ status }) {
        if (status === 'offline')
          setPhotoStatus('offline', '📡 Offline — foto será enviada quando houver sinal');
        if (status === 'done')
          setPhotoStatus('done', '✅ Foto enviada com sucesso');
        if (status === 'failed')
          setPhotoStatus('failed', '⚠️ Falha no envio', true); // showRetry=true
      }
    });
  });
}

function setPhotoStatus(type, msg, showRetry = false) {
  const el = document.getElementById('photo-status');
  if (!el) return;
  el.style.display = 'block';
  el.setAttribute('data-type', type);
  el.innerHTML = msg + (showRetry
    ? ' <button onclick="retryPhotoUpload()" class="retry-btn">↺ Tentar novamente</button>'
    : '');
}
public/pwa/service-worker.js (trecho Background Sync) JS · Service Worker 
/**
 * Background Sync: quando o dispositivo recupera conexão,
 * o Service Worker processa automaticamente a fila de uploads pendentes.
 * O usuário não precisa reabrir o app — acontece em segundo plano.
 */

self.addEventListener('sync', (event) => {
  if (event.tag === 'photo-upload-queue') {
    event.waitUntil(processPhotoQueue());
  }
});

async function processPhotoQueue() {
  const pending = await OfflineQueue.getPending();

  for (const item of pending) {
    if (item.attempts >= 3) {
      // Desiste após 3 tentativas — notifica o usuário via Notification API
      await OfflineQueue.updateStatus(item.id, 'failed_final');
      await notifyUser('Foto não enviada', 'Abra o app para reenviar manualmente.');
      continue;
    }

    try {
      await OfflineQueue.updateStatus(item.id, 'uploading', {
        attempts: item.attempts + 1
      });

      const fd = new FormData();
      fd.append('photo',        item.photoBlob, item.photoName);
      fd.append('review_token', item.reviewToken);
      fd.append('slug',         item.slug);

      const res = await fetch('/api/media/upload', { method: 'POST', body: fd });

      await OfflineQueue.updateStatus(item.id, res.ok ? 'done' : 'pending');

    } catch {
      await OfflineQueue.updateStatus(item.id, 'pending'); // tentará novamente
    }
  }
}

// Registrar o sync quando o usuário fica offline
self.addEventListener('online', () => {
  navigator.serviceWorker.ready
    .then(reg => reg.sync?.register('photo-upload-queue'));
});
